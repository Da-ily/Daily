=== 예외 케이스 응답 포맷

[source,json]
----
{
    "statusCode": HTTP상태코드,
    "errorCode": "에러코드",
    "msg": "에러 메시지",
    "successful": false
}
----

[source, json]
----
{
    "statusCode": 409,
    "errorCode": "DUPLICATED_USERNAME",

    "successful": false
}
----
